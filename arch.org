To enable 
To swap caps=lock and escape Create the file
~/etc/X11/xorg.conf.d/00-keyboard.conf~ with the following content:
#+begin_src shell
Section "InputClass"
        Identifier "system-keyboard"
        MatchIsKeyboard "on"
        Option "XkbLayout" "us"
        Option "XkbModel" "pc105"
        Option "XkbOptions" "caps:swapescape"
EndSection
#+end_src

To prevent screen tearing. Create the file
~/etc/X11/xorg.conf.d/20-intel.conf~ with the following content:

#+begin_src shell
Section "Device"
Identifier "Intel Graphics"
Driver "intel"
Option "AccelMethod" "sna"
Option "TearFree" "true"
Option "DRI" "3"
EndSection
#+end_src

To enable tap to click, two-finger right click, and natural scrolling,
edit the file ~/usr/share/X11/xorg.conf.d/40-libinput.conf~ by finding
the secion with Identifier ~libinput touchscreen catchall~ and adding
the options
#+begin_src shell
        Option "Tapping" "on"
        Option "ClickMethod" "clickfinger"
        Option "NaturalScrolling" "true"
#+end_src

To start exwm initially

#+begin_src shell
#!/bin/sh
userresources=$HOME/.Xresources
usermodmap=$HOME/.Xmodmap
sysresources=/etc/X11/xinit/.Xresources
sysmodmap=/etc/X11/xinit/.Xmodmap

# merge in defaults and keymaps
if [ -f $sysresources ]; then
    xrdb -merge $sysresources
fi

if [ -f $sysmodmap ]; then
    xmodmap $sysmodmap
fi

if [ -f "$userresources" ]; then
    xrdb -merge "$userresources"
fi

if [ -f "$usermodmap" ]; then
    xmodmap "$usermodmap"
fi

# start some nice programs
if [ -d /etc/X11/xinit/xinitrc.d ] ; then
 for f in /etc/X11/xinit/xinitrc.d/?*.sh ; do
  [ -x "$f" ] && . "$f"
 done
 unset f
fi

xrdb -merge ~/.Xresources
exec emacs -mm --debug-init
#+end_src

Automatically execute startx after login (at least in the first terminal)
#+begin_src shell
#
# ~/.bash_profile
#
[[ -f ~/.bashrc ]] && . ~/.bashrc
if [[ -z $DISPLAY ]] && [[ $(tty) = /dev/tty1 ]]; then
    startx
fi
#+end_src
